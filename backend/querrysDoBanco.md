### TB_CLIENTE

```c
    CREATE TABLE TB_CLIENTE(

    SENHA_CLI VARCHAR (20) NOT NULL,
```

###  TB_PRODUTO

```c
      CREATE TABLE TB_PRODUTO(  
    ID_PROD INT AUTO_INCREMENT NOT NULL PRIMARY KEY,
    NOME_PROD VARCHAR (80) NOT NULL,
    DESC_PROD TEXT NOT NULL,
    PRECO_PROD DECIMAL (7,2) NOT NULL,
    MARCA_PROD VARCHAR (30),
    MODELO_PROD VARCHAR (30),
    STATUS_PROD INT (2) NOT NULL,
    QTD_PROD INT NOT NULL,
    ID_VEND INT NOT NULL,
    FOREIGN KEY(ID_VEND) REFERENCES TB_VENDEDOR(ID_VEND),
     ID_CAT INT NOT NULL,
    FOREIGN KEY(ID_CAT) REFERENCES TB_CATEGORIA_PRODUTO(ID_CAT)
    )
    

```

###  TB_PEDIDO_VENDA
